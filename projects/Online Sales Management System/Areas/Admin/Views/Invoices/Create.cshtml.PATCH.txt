<!-- ===== ADD/REPLACE the following JS inside @section Scripts ===== -->
<script>
document.addEventListener('DOMContentLoaded', () => {
  const select = document.getElementById('customerSelect');
  const btn = document.getElementById('customerInfoBtn');
  const pop = document.getElementById('customerInfoPopover');

  const n = document.getElementById('ciName');
  const p = document.getElementById('ciPhone');
  const e = document.getElementById('ciEmail');
  const a = document.getElementById('ciAddress');

  function update() {
    const opt = select.options[select.selectedIndex];
    if (!opt || opt.dataset.walkin === "1") {
      btn.style.display = 'none';
      pop.style.display = 'none';
      return;
    }

    btn.style.display = 'inline-flex';
    n.textContent = opt.dataset.name || '';
    p.innerHTML = opt.dataset.phone ? `<b>📞</b> ${opt.dataset.phone}` : '';
    e.innerHTML = opt.dataset.email ? `<b>✉️</b> ${opt.dataset.email}` : '';
    a.innerHTML = opt.dataset.address ? `<b>📍</b> ${opt.dataset.address}` : '';
  }

  function place() {
    const r = btn.getBoundingClientRect();
    pop.style.left = Math.min(r.left, window.innerWidth - pop.offsetWidth - 12) + 'px';
    pop.style.top = (r.bottom + 8) + 'px';
  }

  let t;
  btn.addEventListener('mouseenter', () => {
    clearTimeout(t);
    pop.style.display = 'block';
    place();
  });
  btn.addEventListener('mouseleave', () => t = setTimeout(() => pop.style.display = 'none', 150));
  pop.addEventListener('mouseenter', () => clearTimeout(t));
  pop.addEventListener('mouseleave', () => t = setTimeout(() => pop.style.display = 'none', 150));

  select.addEventListener('change', update);
  update();
});
</script>
